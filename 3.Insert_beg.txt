#include<stdio.h>
struct node
{
    int data;
    struct node *next;
}*head;
void insert_beg()
{

    int data;
    scanf("%d",&data);
    struct node *temp;
    temp=(struct node*)malloc(sizeof(struct node));
    temp->data=data;
    temp->next=NULL;
    temp->next=head;
    head=temp;
}
void append()
{
    int data;
    scanf("%d",&data);
    struct node *temp;
    temp=(struct node*)malloc(sizeof(struct node));
    temp->data=data;
    temp->next=NULL;
    if(head==NULL)
    {
        head=temp;
    }
    else
    {
        struct node *p;
        p=head;
        while(p->next!=NULL)
        {
            p=p->next;
        }
        p->next=temp;
    }
}
void display()
{
    struct node *p;
    p=head;
    while(p!=NULL)
    {
        printf("%d ",p->data);
        p=p->next;
    }
    printf("\n");
}
int main()
{
    int i,n,r;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        append();
    }
    display();
    scanf("%d",&r);
    for(i=1;i<=r;i++)
    {
        insert_beg();
    }
    display();

}
